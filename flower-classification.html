base_dir = "/content/flowers"

# Load image labels from the .mat file
labels_path = os.path.join(base_dir, "imagelabels.mat")
labels_data = scipy.io.loadmat(labels_path)
labels = labels_data["labels"][0]  # Labels are 1-indexed (1-102)

# Group images by class using a dictionary (class label  list of image indices)
class_images = defaultdict(list)
for i, label in enumerate(labels):
    if label == 98:  # Merge Petunia (98  51) for dataset consistency
        label = 51
    class_images[label].append(i)  # Store image index under the correct label

# Function to split image indices into training, validation, and test sets
def split_images(image_indices):
    np.random.shuffle(image_indices)  # Shuffle to randomize distribution

    num_images = len(image_indices)
    train_size = int(num_images * 0.70)  # 70% of images for training
    valid_size = int(num_images * 0.20)  # 20% for validation
    test_size = num_images - train_size - valid_size  # Remaining for testing

    return (
        image_indices[:train_size],  # Training set
        image_indices[train_size:train_size + valid_size],  # Validation set
        image_indices[train_size + valid_size:]  # Test set
    )

# Function to move images to their respective class folders inside train/valid/test
def move_images(image_indices, split_folder):
    for i in tqdm(image_indices, desc=f"Processing {split_folder}"):  # Display progress bar
        image_name = f"image_{i+1:05d}.jpg"  # Image filenames follow "image_XXXXX.jpg" format
        label = labels[i]  # Retrieve class label

        if label == 98:  # Ensure consistency with Petunia class merge
            label = 51

        label_folder = os.path.join(split_folder, str(label))  # Create class subfolder
        os.makedirs(label_folder, exist_ok=True)  # Ensure the folder exists

        src = os.path.join(image_dir, image_name)  # Source image path
        dst = os.path.join(label_folder, image_name)  # Destination path in the split folder

        if os.path.exists(src):  # Move the image only if it exists
            shutil.move(src, dst)

# Process each class separately and distribute images
for class_label, image_indices in class_images.items():
    train_idx, valid_idx, test_idx = split_images(image_indices)  # Split data

    move_images(train_idx, train_dir)  # Move training images
    move_images(valid_idx, valid_dir)  # Move validation images
    move_images(test_idx, test_dir)  # Move test images

# Remove the original image folder after all images are moved
shutil.rmtree(image_dir)
print(" Dataset is now correctly split into train/valid/test with balanced class distribution!")

# Training loop

    running_loss = 0  # Accumulate training loss over the epoch
    model.train()  # Set model to training mode (enables dropout, batch norm, etc.)

    # Loop over training batches
    for inputs, labels in dataloaders['train']:
        steps += 1
        inputs, labels = inputs.to(device), labels.to(device)  # Move data to GPU/CPU

        optimizer.zero_grad()  # Reset accumulated gradients
        outputs = model(inputs)  # Forward pass
        loss = criterion(outputs, labels)  # Compute loss
        loss.backward()  # Backpropagation (calculate gradients)
        optimizer.step()  # Update model parameters

        running_loss += loss.item()  # Track total loss for the batch

        scheduler.step() # Adjust learning rate

    # After each epoch, evaluate the model
    model.eval()  # Switch to evaluation mode (disables dropout, batch norm)
    valid_loss = 0
    accuracy = 0

    with torch.no_grad():  # Disable gradient tracking for validation
        for inputs, labels in dataloaders['valid']:
            inputs, labels = inputs.to(device), labels.to(device)  # Move to GPU/CPU

            outputs = model(inputs)  # Forward pass
            loss = criterion(outputs, labels)  # Compute validation loss
            valid_loss += loss.item()

            # Compute accuracy
            _, preds = torch.max(outputs, 1)  # Get class predictions
            accuracy += torch.sum(preds == labels).item()  # Count correct predictions

    # Compute average validation loss and accuracy per epoch
    train_loss = running_loss / len(dataloaders['train'])  # Normalize training loss
    valid_loss /= len(dataloaders['valid'])  # Normalize validation loss
    accuracy /= len(image_datasets['valid'])  # Normalize accuracy

          f"Valid Accuracy: {accuracy:.3f}")

    # Store loss and accuracy per epoch
    train_losses_per_epoch.append(train_loss)
    valid_losses_per_epoch.append(valid_loss)
